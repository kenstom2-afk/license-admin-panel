1. Validate License (Activation)
POST /api/client/validate
Content-Type: application/json

{
    "license_key": "LIC-XXXX-XXXX",
    "hwid": "device_unique_id",
    "device_info": "Device Information"
}

{
    "valid": true/false,
    "message": "Status message",
    "expires_at": "2024-12-31T23:59:59"
}

POST /api/client/check
Content-Type: application/json

{
    "license_key": "LIC-XXXX-XXXX",

}
API Testing
Code Examples
import requests

api_url = "''' + window.location.origin + '''/api/client/validate"
license_key = "YOUR_LICENSE_KEY"
hwid = "device_unique_id"

response = requests.post(api_url, json={
    "license_key": license_key,
    "hwid": hwid,
    "device_info": "Python Application"
})

if response.json().get("valid"):
    print("License valid!")
else:
    print("License invalid:", response.json().get("message"))
